98 inline double dist_to_region_sqr( const double& qi, const Region& r, unsigned int global_offset)
99 {
100 double r_min_est = r.min_dp[1] * (global_offset - r.min_i) + r.min_dp[0];
101 double r_max_est = r.max_dp[1] * (global_offset - r.min_i) + r.max_dp[0];
102
103 if (qi < r_min_est) return (qi-r_min_est)*(qi-r_min_est);
104 if (qi > r_max_est) return (qi-r_max_est)*(qi-r_max_est);
105 return 0.0;
106 }
