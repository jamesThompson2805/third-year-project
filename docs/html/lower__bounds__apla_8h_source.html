29 inline double region_area( const Region& r) {
30 double width = r.max_i - r.min_i;
31 double min_a = r.min_dp[0];
32 double min_b= r.min_dp[0] + width*r.min_dp[1];
33 double max_a = r.max_dp[0];
34 double max_b= r.max_dp[0] + width*r.max_dp[1];
35 return (min(max_a,max_b) - max(min_a,min_b)) * width
36 + (max(max_a,max_b) - min(max_a,max_b)) * width * 0.5
37 + (max(min_a,min_b) - min(min_a,min_b)) * width * 0.5;
38 }
39 inline Region region_merge( const Region& r1, const Region& r2)
40 {
41 int min_i = min(r1.min_i, r2.min_i);
42 int max_i = max(r1.max_i, r2.max_i);
43 if (max_i == min_i) return { { min(r1.min_dp[0],r2.min_dp[0]),0}
44 , (unsigned int) min_i
45 , { max(r1.max_dp[0],r2.max_dp[0]),0}
46 , (unsigned int) max_i };
47
48 int min_i_rel_r1 = min_i - r1.min_i;
49 int max_i_rel_r1 = max_i - r1.min_i;
50 int min_i_rel_r2 = min_i - r2.min_i;
51 int max_i_rel_r2 = max_i - r2.min_i;
52
53 double r1_lmax = r1.max_dp[1]*min_i_rel_r1 + r1.max_dp[0];
54 double r1_rmax = r1.max_dp[1]*max_i_rel_r1 + r1.max_dp[0];
55 double r2_lmax = r2.max_dp[1]*min_i_rel_r2 + r2.max_dp[0];
56 double r2_rmax = r2.max_dp[1]*max_i_rel_r2 + r2.max_dp[0];
57
58 double r1_lmin = r1.min_dp[1]*min_i_rel_r1 + r1.min_dp[0];
59 double r1_rmin = r1.min_dp[1]*max_i_rel_r1 + r1.min_dp[0];
60 double r2_lmin = r2.min_dp[1]*min_i_rel_r2 + r2.min_dp[0];
61 double r2_rmin = r2.min_dp[1]*max_i_rel_r2 + r2.min_dp[0];
62
63 double r_lmax = max( max(r1_lmax, r2_lmax), max(r1_lmin, r2_lmin) );
64 double r_lmin = min( min(r1_lmax, r2_lmax), min(r1_lmin, r2_lmin) );
65 double r_rmax = max( max(r1_rmax, r2_rmax), max(r1_rmin, r2_rmin) );
66 double r_rmin = min( min(r1_rmax, r2_rmax), min(r1_rmin, r2_rmin) );
67 return { {r_lmin, (r_rmin-r_lmin)/double(max_i-min_i)}, (unsigned int) min_i , {r_lmax, (r_rmax-r_lmax)/double(max_i-min_i)}, (unsigned int) max_i};
68 }

70 inline double dist_to_region_sqr( const double& qi, const Region& r, unsigned int global_offset)
71 {
72 double r_min_est = r.min_dp[1] * (global_offset - r.min_i) + r.min_dp[0];
73 double r_max_est = r.max_dp[1] * (global_offset - r.min_i) + r.max_dp[0];
74
75 if (qi < r_min_est) return (qi-r_min_est)*(qi-r_min_est);
76 if (qi > r_max_est) return (qi-r_max_est)*(qi-r_max_est);
77 return 0.0;
78 }
79 inline double dist_to_regions_sqr( const double& qi, const Region* const rstart, const Region* const rend, unsigned int global_offset)
80 {
81 auto min_f = [&](const double& min_d, const Region& r){
82 if (double rd = dist_to_region_sqr(qi,r,global_offset); min_d < 0 || rd < min_d)
83 return rd;
84 return min_d;
85 };
86 return std::accumulate(rstart, rend+1, -1.0, min_f);
87 }

129 inline Region ptrs_to_region(const double* const start, const double* const end, unsigned int g_start_i)
130 {
131 if (start == end) return { {start[0], 0}, g_start_i, {start[0], 0}, g_start_i };
132 DoublePair dp = pla::regression(start, end);
133 if (start+1 == end) return { {dp[0], dp[1]}, g_start_i, {dp[0], dp[1]}, g_start_i+1 };
134 //std::cout << " dp " << dp[0] << " : " << dp[1] << " width : " << end - start << std::endl;
135 int max_i = 0, min_i = 0;
136 double max_v = -1, min_v = -1;
137 for (int i=0; i<=end-start; i++) {
138 double dist_to_line_sqr = (dp[1]*i - start[i] +dp[0])*(dp[1]*i - start[i] +dp[0]) / std::abs( dp[1]*dp[1] + 1 ) ;
139 //std::cout << " index : " << g_start_i + i <<" val " << start[i] << " dist to line " << dist_to_line_sqr << std::endl;
140 if (dist_to_line_sqr > max_v && start[i] >= dp[0]+dp[1]*i) {
141 max_i = i;
142 max_v = dist_to_line_sqr;
143 }
144 if (dist_to_line_sqr > min_v && start[i] <= dp[0]+dp[1]*i) {
145 min_i = i;
146 min_v = dist_to_line_sqr;
147 }
148 }
149 //std::cout << " chose min i : " << min_i << ", max i : " << max_i << std::endl;
150 int width = end - start;
151 Region ret = { {start[min_i]-dp[1]*min_i, dp[1]}
152 , g_start_i

154 , g_start_i+width };
155 for (int i=0; i<=end-start; i++) {
156 //std::cout <<" index : " << i+g_start_i << " dist to region " << dist_to_region_sqr(start[i], ret, g_start_i+i) << std::endl;
157 }
